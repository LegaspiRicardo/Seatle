Consultas

Horario designado a cada profesor

SELECT nombres, id_horario_profe FROM profesor, horario_profe WHERE profesor.id_profesor = horario_profe.id_horario_profe;



Tabla para ver que grupo le corresponde a cada profesor

SELECT nombres, apellidos, nivel
FROM profesor_grupo AS pg INNER JOIN grupo AS g ON pg.id_grupo=g.id_grupo INNER JOIN profesor AS p ON pg.id_profesor=p.id_profesor;



Tabla para ver las clases grupales relacionadas a x profesor

SELECT nombres, apellidos, nivel, dia, hora_inicio, hora_fin
FROM profesor_grupo AS pg INNER JOIN grupo AS g ON pg.id_grupo=g.id_grupo INNER JOIN profesor AS p ON pg.id_profesor=p.id_profesor INNER JOIN clases_grupo
WHERE p.nombres='Zoe'  
ORDER BY `clases_grupo`.`dia` ASC;
