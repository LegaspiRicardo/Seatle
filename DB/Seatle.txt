CREATE TABLE alumno (
id_alumno INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
nombres  VARCHAR(60) NOT NULL,
apellido_pat VARCHAR(60) NOT NULL,
apellido_mat VARCHAR(60) NOT NULL,
calle VARCHAR(60) NOT NULL,
numero VARCHAR(20) NOT NULL,
colonia VARCHAR(60) NOT NULL,
cp VARCHAR(20) NOT NULL,
fecha_nac DATE NOT NULL,
sexo VARCHAR(20),
fecha_ing DATE NOT NULL,
id_tutor INT NOT NULL,
tipo_clase VARCHAR(60),
nivel VARCHAR(60),
comentarios  VARCHAR(250),
alberca VARCHAR(60) NOT NULL,
status VARCHAR(60) NOT NULL);

CREATE TABLE tutor (
id_tutor INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
nombre  VARCHAR(60) NOT NULL,
correo  VARCHAR(60),
telefono  VARCHAR(20) NOT NULL);

CREATE TABLE mensualidad  (
id_mensualidad INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
id_alumno INT NOT NULL,
tipo_clase VARCHAR(60) NOT NULL,
precio_clase VARCHAR(60) NOT NULL,
num_clases INT(20) NOT NULL,
total VARCHAR(20) NOT NULL);

CREATE TABLE profesor  (
id_profesor INT PRIMARY KEY AUTO_INCREMENT NOT NULL,

fecha_nac DATE NOT NULL,
fecha_ing DATE NOT NULL,
alberca  VARCHAR(60) NOT NULL,

status VARCHAR(20),
comentarios  VARCHAR(250));

CREATE TABLE horario_profe (
id_profesor INT NOT NULL,

CREATE TABLE grupo (
